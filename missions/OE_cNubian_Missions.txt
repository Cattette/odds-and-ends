gen_christian_nubia_2_unlocked = {
	slot = 2
	generic = yes
	ai = yes
	potential_on_load = {
	}
	potential = {
		NOT = { map_setup = map_setup_random }
		religion_group = christian
		culture_group = sudanese
		has_country_flag = is_nubian_leader
	}
	has_country_shield = yes

	nubian_allies = {
		icon = mission_galleys_in_port
		position = 2
		required_missions = { leader_of_nubian_copts }
		provinces_to_highlight = {
			OR = {
				owned_by = MAK
				owned_by = ALO
			}
			NOT = { owned_by = root }
		}
		trigger = {
			OR = {
				AND = {
					if = {
						limit = { tag = MAK}
						reverse_has_opinion = { who = ALO value = 120 } 
					}
					if = {
						limit = { tag = ALO }
						reverse_has_opinion = { who = MAK value = 120 } 
					}
				}
				AND = {
					if = {
						limit = { tag = MAK}
						ALO = {
							OR = {
								has_opinion_modifier = {
									who = ROOT
									modifier = scornfully_insulted
								}
								has_opinion_modifier = {
									who = ROOT
									modifier = insulted
								}
							}
						}
					}
					if = {
						limit = { tag = ALO }
						MAK = {
							OR = {
								has_opinion_modifier = {
									who = ROOT
									modifier = scornfully_insulted
								}
								has_opinion_modifier = {
									who = ROOT
									modifier = insulted
								}
							}
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = { 
					#ai = no
					OR = {
						reverse_has_opinion = { who = ALO value = 120 }
						reverse_has_opinion = { who = MAK value = 120 } 
					} 
				}
				custom_tooltip = yes
				custom_tooltip = nubian_allies_friend
				country_event = { id = cNubia_events.3 }
				custom_tooltip = no
				custom_tooltip = nubian_allies_enemy
				tooltip = {
					country_event = { id = cNubia_events.2 }
				}
			}
			if = {
				limit = { 
					#ai = no
					OR = {
						MAK = {
							OR = {
								has_opinion_modifier = { who = ROOT modifier = scornfully_insulted }
								has_opinion_modifier = { who = ROOT modifier = insulted }
							}
						}
						ALO = {
							OR = {
								has_opinion_modifier = { who = ROOT modifier = scornfully_insulted }
								has_opinion_modifier = { who = ROOT modifier = insulted }
							}
						}
					} 
				}
				custom_tooltip = no
				custom_tooltip = nubian_allies_friend
				tooltip = { country_event = { id = cNubia_events.3 } }
				custom_tooltip = yes
				custom_tooltip = nubian_allies_enemy
				country_event = { id = cNubia_events.2 }
			}
			else = {
				custom_tooltip = no
				custom_tooltip = nubian_allies_friend
				tooltip = {
					country_event = { id = cNubia_events.3 }
				}
				custom_tooltip = no
				custom_tooltip = nubian_allies_enemy
				tooltip = { 					
					country_event = { id = cNubia_events.2 }
				}
			}
		}
	}
}

			num_of_infantry = 10

		}
		effect = {
